**localStorage**

存储期限: 永久

设置
```
localStorage.setItem("key","value")
```
获取
```
localStorage.getItem("key")
```
移除
```
localStorage.remove("key")
```
全部清除
```
localStorage.clear()
```
---

**sessionStorage**

存储期限: 会话结束

**小案例:待办事项**
```
<!DOCTYPE html>
<html>

<head>
  <meta charset="UTF-8">
  <title>Will</title>
  <style>
    * {
      margin: 0;
      padding: 0;
      user-select: none;
    }

    a {
      text-decoration: none;
    }
    #box{
      width: 400px;
      margin: auto;
      padding: 10px;
      overflow-y: auto; 
    }
    
    #ipt{
      width: 300px;
      height: 20px;
      outline: none;
      margin: 20px;
      text-indent: 2px;
      font-size: 10px;
    }
    
    #wrap{
      width: 350px;
      height: 100%;
      margin: auto
    }

    #wrap ul li{
      border-bottom: 1px solid;
      width: 300px;
      margin-bottom: 12px;
      position: relative;
      cursor: pointer;
    }

    #wrap ul li span{
      display: inline-block;
      position: absolute;
      right: 0px;
      bottom: 0;
      width: 20px;
      height: 20px;
      background: grey;
      color: red;
      line-height: 16px;
      font-size: 20px;
      text-align: center;
    }

    .done{
      color: #cccccb;
      text-decoration: line-through;
    }
  </style>
</head>

<body>
  <div id="box">
    <h2>代办事项</h2>
    <input type="text" name="ipt" id="ipt" placeholder="输入事件,空格确认">
    <div id='wrap'>
      <ul id='list'>

      </ul>
    </div>
  </div>
  <script>
    /*数据类型
    key: toDoList
    value:
    [{
      content: 内容1,
      done: true
    },
    {
      content: 内容2,
      done: false
    }]
    */
    //localStorage.setItem("toDoList", JSON.stringify([{ content: "吃饭", done: false }]))
    const ipt = document.getElementById("ipt")
    const list = document.getElementById("list")
    let data = localStorage.getItem("toDoList")

    if (data === null) {
      localStorage.setItem('toDoList', '[]')
    } else {
      JSON.parse(data).forEach(v => {
        list.innerHTML += `<li class="${v.done ? 'done' : ""}">${v.content}<span>x</span></li>`
      });
    }

    ipt.onkeyup = function (e) {
      let content = this.value.trim()
      //事件兼容
      e = e || window.event
      //不是空格,输入为空返回
      if (e.keyCode !== 13) return
      if (!content) return
      let data = JSON.parse(localStorage.getItem("toDoList"))

      //更新页面
      list.innerHTML += `<li>${content}<span>x</span></li>`

      //更新存储
      data.push({
        content,
        done: false
      })
      localStorage.setItem("toDoList", JSON.stringify(data))

      //清空页面
      this.value = ''
    }

    list.addEventListener("click", function (e) {
      e = e || window.event
      const that = e.target
      let data = JSON.parse(localStorage.getItem("toDoList"))
      e.stopPropagation()

      const tagName = that.tagName.toLowerCase()
      switch (tagName) {
        case "li":
          data.forEach(function (v, i) {
            if (this === list.children[i]) {
              this.classList.toggle('done')
              v.done = !v.done
              localStorage.setItem("toDoList", JSON.stringify(data))
            }
          },that)         
          break;
        case "span":
          //forEach 的第二个参数 this指向
          data.forEach(function (v, i, self) {
            if (this === list.children[i]) {
              list.removeChild(this)
              self.splice(i, 1)
              localStorage.setItem("toDoList", JSON.stringify(data))
            }
          }, that.parentNode)
          break;
        //没有默认事件
      }
    })
  </script>
</body>

</html>

```
